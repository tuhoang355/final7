def count_kmers(read, k):
    
 
    counts = {}
  
    num_kmers = len(read) - k + 1
   
    for i in range(num_kmers):
     
        kmer = read[i:i+k]
      
        if kmer not in counts:
            counts[kmer] = 0

        counts[kmer] += 1
 
    return counts
